214 uint32_t LESENSE_ScanFreqSet(uint32_t refFreq, uint32_t const scanFreq)
215 {
216  uint32_t tmp;
217  uint32_t pcPresc = 0UL; /* Period counter prescaler. */
218  uint32_t clkDiv = 1UL; /* Clock divisor value (2^pcPresc). */
219  uint32_t pcTop = 63UL; /* Period counter top value (max. 63). */
220  uint32_t calcScanFreq; /* Variable for testing the calculation algorithm. */
221 
222 
223  /* If refFreq is set to 0, the currently configured reference clock is
224  * assumed. */
225  if (!refFreq)
226  {
227  refFreq = CMU_ClockFreqGet(cmuClock_LESENSE);
228  }
229 
230  /* Max. value of pcPresc is 128, thus using reference frequency less than
231  * 33554431Hz (33.554431MHz), the frequency calculation in the while loop
232  * below will not overflow. */
233  EFM_ASSERT(refFreq < ((uint32_t)UINT32_MAX / 128UL));
234 
235  /* Sanity check of scan frequency value. */
236  EFM_ASSERT((scanFreq > 0U) && (scanFreq <= refFreq));
237 
238  /* Calculate the minimum necessary prescaler value in order to provide the
239  * biggest possible resolution for setting scan frequency.
240  * Maximum number of calculation cycles is 7 (value of lesenseClkDiv_128). */
241  while ((refFreq / ((uint32_t)scanFreq * clkDiv) > (pcTop + 1UL))
242  && (pcPresc < lesenseClkDiv_128))
243  {
244  ++pcPresc;
245  clkDiv = (uint32_t)1UL << pcPresc;
246  }
247 
248  /* Calculate pcTop value. */
249  pcTop = ((uint32_t)refFreq / ((uint32_t)scanFreq * clkDiv)) - 1UL;
250 
251  /* Clear current PCPRESC and PCTOP settings. Be aware of the effect of
252  * non-atomic Read-Modify-Write on LESENSE->TIMCRTL. */
253  tmp = LESENSE->TIMCTRL & (~_LESENSE_TIMCTRL_PCPRESC_MASK
254  & ~_LESENSE_TIMCTRL_PCTOP_MASK);
255 
256  /* Set new values in tmp while reserving other settings. */
257  tmp |= ((uint32_t)pcPresc << _LESENSE_TIMCTRL_PCPRESC_SHIFT)
258  | ((uint32_t)pcTop << _LESENSE_TIMCTRL_PCTOP_SHIFT);
259 
260  /* Set values in LESENSE_TIMCTRL register. */
261  LESENSE->TIMCTRL = tmp;
262 
263  /* For testing the calculation algorithm. */
264  calcScanFreq = ((uint32_t)refFreq / ((uint32_t)(1UL + pcTop) * clkDiv));
265 
266  return calcScanFreq;
267 }

505  && (chIdx != 1U)
506  && (chIdx != 2U)
507  && (chIdx != 3U))));
508 
509  /* Configure chIdx setup in LESENSE idle phase.
510  * Read-modify-write in order to support reconfiguration during LESENSE
511  * operation. */
512  tmp = (LESENSE->IDLECONF & ~((uint32_t)0x3UL << (chIdx * 2UL)));

524  * alternate excitation usage and interrupt mode on scan channel chIdx in
525  * LESENSE_CHchIdx_INTERACT. */
526  LESENSE->CH[chIdx].INTERACT =
527  ((uint32_t)confCh->exClk << _LESENSE_CH_INTERACT_EXCLK_SHIFT)
528  | ((uint32_t)confCh->sampleClk << _LESENSE_CH_INTERACT_SAMPLECLK_SHIFT)
529  | (uint32_t)confCh->sampleMode
530  | (uint32_t)confCh->intMode
531  | (uint32_t)confCh->chPinExMode
532  | ((uint32_t)confCh->useAltEx << _LESENSE_CH_INTERACT_ALTEX_SHIFT);
533 
534  /* Configure channel specific counter comparison mode, optional result
535  * forwarding to decoder, optional counter value storing and optional result
536  * inverting on scan channel chIdx in LESENSE_CHchIdx_EVAL. */
537  LESENSE->CH[chIdx].EVAL =
538  (uint32_t)confCh->compMode
539  | ((uint32_t)confCh->shiftRes << _LESENSE_CH_EVAL_DECODE_SHIFT)
540  | ((uint32_t)confCh->storeCntRes << _LESENSE_CH_EVAL_STRSAMPLE_SHIFT)
541  | ((uint32_t)confCh->invRes << _LESENSE_CH_EVAL_SCANRESINV_SHIFT);
542 
543  /* Configure analog comparator (ACMP) threshold and decision threshold for
544  * counter separately with the function provided for that. */
545  LESENSE_ChannelThresSet(chIdx,
546  (uint32_t)confCh->acmpThres,
547  (uint32_t)confCh->cntThres);
548 
549  /* Enable/disable interrupts on channel */
550  BUS_RegBitWrite(&(LESENSE->IEN), chIdx, confCh->enaInt);
551 
552  /* Enable/disable CHchIdx pin. */
553  BUS_RegBitWrite(&(LESENSE->ROUTE), chIdx, confCh->enaPin);
554 
555  /* Enable/disable scan channel chIdx. */
556  BUS_RegBitWrite(&(LESENSE->CHEN), chIdx, confCh->enaScanCh);
557 }

577 void LESENSE_AltExConfig(LESENSE_ConfAltEx_TypeDef const *confAltEx)
578 {
579  uint32_t i;
580  uint32_t tmp;
581 
582 
583  /* Configure alternate excitation mapping.
584  * Atomic read-modify-write using BUS_RegBitWrite function in order to
585  * support reconfiguration during LESENSE operation. */
586  BUS_RegBitWrite(&(LESENSE->CTRL),
587  _LESENSE_CTRL_ALTEXMAP_SHIFT,
588  confAltEx->altExMap);
589 
590  switch (confAltEx->altExMap)
591  {
592  case lesenseAltExMapALTEX:
593  /* Iterate through the 8 possible alternate excitation pin descriptors. */
594  for (i = 0U; i < 8U; ++i)
595  {
596  /* Enable/disable alternate excitation pin i.
597  * Atomic read-modify-write using BUS_RegBitWrite function in order to
598  * support reconfiguration during LESENSE operation. */
599  BUS_RegBitWrite(&(LESENSE->ROUTE),
600  (16UL + i),
601  confAltEx->AltEx[i].enablePin);
602 
603  /* Setup the idle phase state of alternate excitation pin i.
604  * Read-modify-write in order to support reconfiguration during LESENSE
605  * operation. */
606  tmp = (LESENSE->ALTEXCONF & ~((uint32_t)0x3UL << (i * 2UL)));
607  tmp |= ((uint32_t)confAltEx->AltEx[i].idleConf << (i * 2UL));
608  LESENSE->ALTEXCONF = tmp;
609 
610  /* Enable/disable always excite on channel i */
611  BUS_RegBitWrite(&(LESENSE->ALTEXCONF),
612  (16UL + i),
613  confAltEx->AltEx[i].alwaysEx);
614  }
615  break;
616 
617  case lesenseAltExMapACMP:
618  /* Iterate through all the 16 alternate excitation channels */
619  for (i = 0U; i < 16U; ++i)
620  {
621  /* Enable/disable alternate ACMP excitation channel pin i. */
622  /* Atomic read-modify-write using BUS_RegBitWrite function in order to
623  * support reconfiguration during LESENSE operation. */
624  BUS_RegBitWrite(&(LESENSE->ROUTE),
625  i,
626  confAltEx->AltEx[i].enablePin);
627  }
628  break;
629  default:
630  /* Illegal value. */
631  EFM_ASSERT(0);
632  break;
633  }
634 }
