71  int result;

81 
82  if( bmpDeviceId != BMP_DEVICE_ID_BMP280 ) {
83  return BMP_ERROR_DEVICE_ID_MISMATCH;
84  }
85 
86  bmp280.bus_write = i2cBusWrite;
87  bmp280.bus_read = i2cBusRead;
88  bmp280.dev_addr = BMP_I2C_BUS_ADDRESS;
89  bmp280.delay_msec = UTIL_delay;
90 
91  result = bmp280_init( &bmp280 );
92 
93  if( result != BMP_OK ) {
94  return result;
95  }
96

98 
99  if( result != BMP_OK ) {
100  return result;
101  }
102

104 
105  if( result != BMP_OK ) {
106  return result;
107  }

251 static uint8_t readRegister( uint8_t addr )
252 {
253 
254  I2C_TransferSeq_TypeDef seq;
255  I2C_TransferReturn_TypeDef ret;
256  uint8_t reg;
257 
258  seq.addr = BMP_I2C_BUS_ADDRESS << 1;
259  seq.flags = I2C_FLAG_WRITE_READ;
260 
261  seq.buf[0].len = 1;
262  seq.buf[0].data = &addr;
263  seq.buf[1].len = 1;
264  seq.buf[1].data = &reg;
265 
266  ret = I2CSPM_Transfer( BMP_I2C_DEVICE, &seq );
267 
268  if( ret != i2cTransferDone ) {
269  return 0;
270  }
271 
272  return reg;
273 
274 }
