411 static bool ReceiveDmaComplete(unsigned int channel,
412  unsigned int sequenceNo,
413  void *userParam)
414 {
415  UARTDRV_Handle_t handle;
416  UARTDRV_Buffer_t *buffer;
417  (void)channel;
418  (void)sequenceNo;
419 
420  handle = (UARTDRV_Handle_t)userParam;
421  GetTailBuffer(handle->rxQueue, &buffer);
422 
423  if (handle->type == uartdrvUartTypeUart
424  && (handle->peripheral.uart->IF & USART_IF_FERR))
425  {
426  buffer->transferStatus = ECODE_EMDRV_UARTDRV_FRAME_ERROR;
427  buffer->itemsRemaining = 0;
428  handle->peripheral.uart->IFC = USART_IFC_FERR;
429  }
430  else if (handle->type == uartdrvUartTypeLeuart
431  && (handle->peripheral.leuart->IF & LEUART_IF_FERR))
432  {
433  buffer->transferStatus = ECODE_EMDRV_UARTDRV_FRAME_ERROR;
434  buffer->itemsRemaining = 0;
435  handle->peripheral.leuart->IFC = LEUART_IFC_FERR;
436  }
437  else if (handle->type == uartdrvUartTypeUart
438  && (handle->peripheral.uart->IF & USART_IF_PERR))
439  {
440  buffer->transferStatus = ECODE_EMDRV_UARTDRV_PARITY_ERROR;
441  buffer->itemsRemaining = 0;
442  handle->peripheral.uart->IFC = USART_IFC_PERR;
443  }
444  else if (handle->type == uartdrvUartTypeLeuart
445  && (handle->peripheral.leuart->IF & LEUART_IF_PERR))
446  {
447  buffer->transferStatus = ECODE_EMDRV_UARTDRV_PARITY_ERROR;
448  buffer->itemsRemaining = 0;
449  handle->peripheral.leuart->IFC = LEUART_IFC_PERR;
450  }
451  else
452  {
453  buffer->transferStatus = ECODE_EMDRV_UARTDRV_OK;
454  buffer->itemsRemaining = 0;
455  }
456 
457  INT_Disable();
458 
459  if (buffer->callback != NULL)
460  {
461  buffer->callback(handle, buffer->transferStatus, buffer->data, buffer->transferCount - buffer->itemsRemaining);
462  }
463  // Dequeue the current tail RX operation, check if more in queue
464  DequeueBuffer(handle->rxQueue, &buffer);
465 
466  if (handle->rxQueue->used > 0)
467  {
468  GetTailBuffer(handle->rxQueue, &buffer);
469  StartReceiveDma(handle, buffer);
470  }
471  else
472  {
473 #if EMDRV_UARTDRV_HW_FLOW_CONTROL_ENABLE
474  handle->fcSelfState = uartdrvFlowControlOff;
475  FcApplyState(handle);
476 #endif
477  handle->rxDmaActive = false;
478 
479  if (handle->fcType != uartdrvFlowControlHwUart)
480  {
481  DisableReceiver(handle);
482  }
483  }
484  INT_Enable();
485  return true;
486 }
