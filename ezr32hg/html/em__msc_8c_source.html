158 {
159 #if defined( _MSC_TIMEBASE_MASK )
160  uint32_t freq, cycles;
161 #endif
162  /* Unlock the MSC */
163  MSC->LOCK = MSC_UNLOCK_CODE;
164  /* Disable writing to the flash */
165  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
166 
167  /* Call SystemCoreClockGet in order to set the global variable SystemCoreClock
168  which is used in MSC_LoadWriteData to make sure the frequency is
169  sufficiently high. If the clock frequency is changed then software is
170  responsible for calling MSC_Init or SystemCoreClockGet in order to set the
171  SystemCoreClock variable to the correct value. */
172  SystemCoreClockGet();
173 
174 #if defined( _MSC_TIMEBASE_MASK )
175  /* Configure MSC->TIMEBASE according to selected frequency */
176  freq = CMU_ClockFreqGet(cmuClock_AUX);
177 
178  /* Timebase 5us is used for the 1/1.2MHz band only. Note that the 1MHz band
179  is tuned to 1.2MHz on newer revisions. */
180  if (freq > 1200000)
181  {
182  /* Calculate number of clock cycles for 1us as base period */
183  freq = (freq * 11) / 10;
184  cycles = (freq / 1000000) + 1;
185 
186  /* Configure clock cycles for flash timing */
187  MSC->TIMEBASE = (MSC->TIMEBASE & ~(_MSC_TIMEBASE_BASE_MASK
188  | _MSC_TIMEBASE_PERIOD_MASK))
189  | MSC_TIMEBASE_PERIOD_1US
190  | (cycles << _MSC_TIMEBASE_BASE_SHIFT);
191  }
192  else
193  {
194  /* Calculate number of clock cycles for 5us as base period */
195  freq = (freq * 5 * 11) / 10;
196  cycles = (freq / 1000000) + 1;
197 
198  /* Configure clock cycles for flash timing */
199  MSC->TIMEBASE = (MSC->TIMEBASE & ~(_MSC_TIMEBASE_BASE_MASK
200  | _MSC_TIMEBASE_PERIOD_MASK))

379 MSC_Status_TypeDef MSC_LoadWriteData(uint32_t* data,
380  uint32_t numWords,
381  MSC_WriteStrategy_Typedef writeStrategy)
382 {
383  uint32_t timeOut;
384  uint32_t wordIndex;
385  uint32_t wordsPerDataPhase;
386  MSC_Status_TypeDef retval = mscReturnOk;
387 #if !defined( _EFM32_GECKO_FAMILY )
388  uint32_t irqState;
389 #endif
390 
391 #if defined(_MSC_WRITECTRL_LPWRITE_MASK) && defined(_MSC_WRITECTRL_WDOUBLE_MASK)
392  /* If LPWRITE (Low Power Write) is NOT enabled, set WDOUBLE (Write Double word) */
393  if (!(MSC->WRITECTRL & MSC_WRITECTRL_LPWRITE))
394  {
395  /* If the number of words to be written are odd, we need to align by writing
396  a single word first, before setting the WDOUBLE bit. */
397  if (numWords & 0x1)
398  {
399  /* Wait for the MSC to become ready for the next word. */
400  timeOut = MSC_PROGRAM_TIMEOUT;
401  while ((!(MSC->STATUS & MSC_STATUS_WDATAREADY)) && (timeOut != 0))
402  {
403  timeOut--;
404  }
405  /* Check for timeout */
406  if (timeOut == 0)
407  {
408  return mscReturnTimeOut;
409  }
410  /* Clear double word option, in order to write the initial single word. */
411  MSC->WRITECTRL &= ~MSC_WRITECTRL_WDOUBLE;
412  /* Write first data word. */
413  MSC->WDATA = *data++;
414  MSC->WRITECMD = MSC_WRITECMD_WRITEONCE;
415 
416  /* Wait for the operation to finish. It may be required to change the WDOUBLE
417  config after the initial write. It should not be changed while BUSY. */
418  timeOut = MSC_PROGRAM_TIMEOUT;
419  while((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
420  {
421  timeOut--;
422  }
423  /* Check for timeout */
424  if (timeOut == 0)
425  {
426  return mscReturnTimeOut;
427  }
428  /* Subtract this initial odd word for the write loop below */
429  numWords -= 1;
430  retval = mscReturnOk;
431  }
432  /* Now we can set the double word option in order to write two words per
433  data phase. */
434  MSC->WRITECTRL |= MSC_WRITECTRL_WDOUBLE;
435  wordsPerDataPhase = 2;
436  }
437  else
438 #endif /* defined( _MSC_WRITECTRL_LPWRITE_MASK ) && defined( _MSC_WRITECTRL_WDOUBLE_MASK ) */
439  {
440  wordsPerDataPhase = 1;
441  }
442 
443  /* Write the rest as double word write if wordsPerDataPhase == 2 */
444  if (numWords > 0)
445  {
446  /**** Write strategy: mscWriteIntSafe ****/
447  if (writeStrategy == mscWriteIntSafe)
448  {
449  /* Requires a system core clock at 1MHz or higher */
450  EFM_ASSERT(SystemCoreClock >= 1000000);
451  wordIndex = 0;
452  while(wordIndex < numWords)
453  {
454  MSC->WDATA = *data++;
455  wordIndex++;
456  if (wordsPerDataPhase == 1)
457  {
458  MSC->WRITECMD = MSC_WRITECMD_WRITEONCE;
459  }
460  else if (wordsPerDataPhase == 2)
461  {
462  while (!(MSC->STATUS & MSC_STATUS_WDATAREADY));
463  MSC->WDATA = *data++;
464  wordIndex++;
465 
466  /* Trigger double write. Platform 1 and 2
467  have different trigger behavior for
468  double word write as described in the
469  reference manual for MSC_WRITECMD_WRITEONCE
470  and WRITETRIG. */
471 #if defined(_SILICON_LABS_32B_PLATFORM_1)
472  MSC->WRITECMD = MSC_WRITECMD_WRITEONCE;
473 #else
474  MSC->WRITECMD = MSC_WRITECMD_WRITETRIG;
475 #endif
476  }
477  else
478  {
479  /* Not supported. */
480  EFM_ASSERT(false);
481  }
482 
483  /* Wait for the transaction to finish. */
484  timeOut = MSC_PROGRAM_TIMEOUT;
485  while ((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
486  {
487  timeOut--;
488  }
489  /* Check for timeout */
490  if (timeOut == 0)
491  {
492  retval = mscReturnTimeOut;
493  break;
494  }
495 #if defined( _EFM32_GECKO_FAMILY )
496  MSC->ADDRB += 4;
497  MSC->WRITECMD = MSC_WRITECMD_LADDRIM;
498 #endif
499  }
500  }
501 
502  /**** Write strategy: mscWriteFast ****/
503  else
504  {
505 #if defined( _EFM32_GECKO_FAMILY )
506  /* Gecko does not have auto-increment of ADDR. */
507  EFM_ASSERT(false);
508 #else
509  /* Requires a system core clock at 14MHz or higher. */
510  EFM_ASSERT(SystemCoreClock >= 14000000);
511 
512  /*
513  * Protect from interrupts to be sure to satisfy the us timinig
514  * needs of the MSC flash programming state machine.
515  */
516  irqState = __get_PRIMASK();
517  __disable_irq();
518 
519  wordIndex = 0;
520  while(wordIndex < numWords)
521  {
522  /* Wait for the MSC to be ready for the next word. */
523  while (!(MSC->STATUS & MSC_STATUS_WDATAREADY))
524  {
525  /* If the write to MSC->WDATA below missed the 30us timeout and the
526  following MSC_WRITECMD_WRITETRIG command arrived while
527  MSC_STATUS_BUSY is 1, then the MSC_WRITECMD_WRITETRIG could be ignored by
528  the MSC. In this case, MSC_STATUS_WORDTIMEOUT is set to 1
529  and MSC_STATUS_BUSY is 0. A new trigger is therefore needed here to
530  complete write of data in MSC->WDATA.
531  If WDATAREADY became high since entry into this loop, exit and continue
532  to the next WDATA write.
533  */
534  if ((MSC->STATUS & (MSC_STATUS_WORDTIMEOUT
535  | MSC_STATUS_BUSY
536  | MSC_STATUS_WDATAREADY))
537  == MSC_STATUS_WORDTIMEOUT)
538  {
539  MSC->WRITECMD = MSC_WRITECMD_WRITETRIG;
540  }
541  }
542  MSC->WDATA = *data;
543  if ((wordsPerDataPhase == 1)
544  || ((wordsPerDataPhase == 2) && (wordIndex & 0x1)))
545  {
546  MSC->WRITECMD = MSC_WRITECMD_WRITETRIG;
547  }
548  data++;
549  wordIndex++;
550  }
551 
552  if (irqState == 0)
553  {
554  /* Restore previous interrupt state. */
555  __enable_irq();
556  }
557 
558  /* Wait for the transaction to finish. */
559  timeOut = MSC_PROGRAM_TIMEOUT;
560  while ((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
561  {
562  timeOut--;
563  }
564  /* Check for timeout */
565  if (timeOut == 0)
566  {
567  retval = mscReturnTimeOut;
568  }
569 #endif
570  } /* writeStrategy */
571  }
572 
573 #if defined( _MSC_WRITECTRL_WDOUBLE_MASK )
574  /* Clear double word option, which should not be left on when returning. */
575  MSC->WRITECTRL &= ~MSC_WRITECTRL_WDOUBLE;
576 #endif
577 
578  return retval;
579 }

598 MSC_Status_TypeDef MSC_WriteWordI(uint32_t *address,
599  void const *data,
600  uint32_t numBytes,
601  MSC_WriteStrategy_Typedef writeStrategy)
602 {
603  uint32_t wordCount;
604  uint32_t numWords;
605  uint32_t pageWords;
606  uint32_t* pData;
607  MSC_Status_TypeDef retval = mscReturnOk;
608 
609  /* Check alignment (Must be aligned to words) */
610  EFM_ASSERT(((uint32_t) address & 0x3) == 0);
611 
612  /* Check number of bytes. Must be divisable by four */
613  EFM_ASSERT((numBytes & 0x3) == 0);
614 
615  /* Enable writing to the MSC */
616  MSC->WRITECTRL |= MSC_WRITECTRL_WREN;
617 
618  /* Convert bytes to words */
619  numWords = numBytes >> 2;
620  EFM_ASSERT(numWords > 0);
621 
622  /* The following loop splits the data into chunks corresponding to flash pages.
623  The address is loaded only once per page, because the hardware automatically
624  increments the address internally for each data load inside a page. */
625  for (wordCount = 0, pData = (uint32_t *)data; wordCount < numWords; )
626  {
627  /* First we load address. The address is auto-incremented within a page.
628  Therefore the address phase is only needed once for each page. */
629  retval = MSC_LoadVerifyAddress(address + wordCount);
630  if (mscReturnOk != retval)
631  {
632  return retval;
633  }
634  /* Compute the number of words to write to the current page. */
635  pageWords =
636  (FLASH_PAGE_SIZE -
637  (((uint32_t) (address + wordCount)) & (FLASH_PAGE_SIZE - 1)))
638  / sizeof(uint32_t);
639  if (pageWords > numWords - wordCount)
640  {
641  pageWords = numWords - wordCount;
642  }
643  /* Now write the data in the current page. */
644  retval = MSC_LoadWriteData(pData, pageWords, writeStrategy);
645  if (mscReturnOk != retval)
646  {
647  break;
648  }
649  wordCount += pageWords;
650  pData += pageWords;
651  }
652 
653 #if defined( ERRATA_FIX_FLASH_E201_EN )
654  /* Fix for errata FLASH_E201 - Potential program failure after Power On.
655  *
656  * Check if the first word was programmed correctly. If a failure is detected
657  * then retry programming of the first word.
658  *
659  * Full description of errata can be found in the errata document */
660  pData = (uint32_t *) data;
661  if (*address != *pData)
662  {
663  retval = MSC_LoadVerifyAddress(address);
664  if (mscReturnOk == retval)
665  {
666  retval = MSC_LoadWriteData(pData, 1, writeStrategy);
667  }
668  }
669 #endif
670 
671  /* Disable writing to the MSC */
672  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
673 
674 #if defined( _MSC_WRITECTRL_WDOUBLE_MASK )
675 #if ( WORDS_PER_DATA_PHASE == 2 )
676  /* Turn off double word write cycle support. */
677  MSC->WRITECTRL &= ~MSC_WRITECTRL_WDOUBLE;
678 #endif
679 #endif
680 
681  return retval;
682 }

714 MSC_Status_TypeDef MSC_ErasePage(uint32_t *startAddress)
715 {
716  uint32_t timeOut = MSC_PROGRAM_TIMEOUT;
717 
718  /* Address must be aligned to pages */
719  EFM_ASSERT((((uint32_t) startAddress) & (FLASH_PAGE_SIZE - 1)) == 0);
720 
721  /* Enable writing to the MSC */
722  MSC->WRITECTRL |= MSC_WRITECTRL_WREN;
723 
724  /* Load address */
725  MSC->ADDRB = (uint32_t)startAddress;
726  MSC->WRITECMD = MSC_WRITECMD_LADDRIM;
727 
728  /* Check for invalid address */
729  if (MSC->STATUS & MSC_STATUS_INVADDR)
730  {
731  /* Disable writing to the MSC */
732  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
733  return mscReturnInvalidAddr;
734  }
735  /* Check for write protected page */
736  if (MSC->STATUS & MSC_STATUS_LOCKED)
737  {
738  /* Disable writing to the MSC */
739  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
740  return mscReturnLocked;
741  }
742  /* Send erase page command */
743  MSC->WRITECMD = MSC_WRITECMD_ERASEPAGE;
744 
745  /* Wait for the erase to complete */
746  while ((MSC->STATUS & MSC_STATUS_BUSY) && (timeOut != 0))
747  {
748  timeOut--;
749  }
750  if (timeOut == 0)
751  {
752  /* Disable writing to the MSC */
753  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
754  return mscReturnTimeOut;
755  }
756  /* Disable writing to the MSC */
757  MSC->WRITECTRL &= ~MSC_WRITECTRL_WREN;
758  return mscReturnOk;
759 }
