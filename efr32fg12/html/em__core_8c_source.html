561 void CORE_YieldNvicMask(const CORE_nvicMask_t *enable)
562 {
563  CORE_nvicMask_t nvicMask;
564 
565  // Get current NVIC enable mask.
566  CORE_CRITICAL_SECTION(
567  nvicMask = *(CORE_nvicMask_t*)&NVIC->ISER[0];
568  )
569 
570  // Make a mask with bits set for those interrupts that are currently
571  // disabled but are set in the enable mask.
572 #if (CORE_NVIC_REG_WORDS == 1)
573  nvicMask.a[0] &= enable->a[0];
574  nvicMask.a[0] = ~nvicMask.a[0] & enable->a[0];
575 
576  if (nvicMask.a[0] != 0) {
577 
578 #elif (CORE_NVIC_REG_WORDS == 2)
579  nvicMask.a[0] &= enable->a[0];
580  nvicMask.a[1] &= enable->a[1];
581  nvicMask.a[0] = ~nvicMask.a[0] & enable->a[0];
582  nvicMask.a[1] = ~nvicMask.a[1] & enable->a[1];
583 
584  if ((nvicMask.a[0] != 0) || (nvicMask.a[1] != 0)) {
585 
586 #elif (CORE_NVIC_REG_WORDS == 3)
587  nvicMask.a[0] &= enable->a[0];
588  nvicMask.a[1] &= enable->a[1];
589  nvicMask.a[2] &= enable->a[2];
590  nvicMask.a[0] = ~nvicMask.a[0] & enable->a[0];
591  nvicMask.a[1] = ~nvicMask.a[1] & enable->a[1];
592  nvicMask.a[2] = ~nvicMask.a[2] & enable->a[2];
593 
594  if ((nvicMask.a[0] != 0) || (nvicMask.a[1] != 0) || (nvicMask.a[2] != 0)) {
595 #endif
596 
597  // Enable previously disabled interrupts.
598  *(CORE_nvicMask_t*)&NVIC->ISER[0] = nvicMask;
599 
600  // Disable those interrupts again.
601  *(CORE_nvicMask_t*)&NVIC->ICER[0] = nvicMask;
602  }
603 }

660 SL_WEAK bool CORE_IrqIsBlocked(IRQn_Type irqN)
661 {
662  uint32_t irqPri, activeIrq;
663 
664 #if (__CORTEX_M >= 3)
665  uint32_t basepri;
666 
667  EFM_ASSERT((irqN >= MemoryManagement_IRQn) && (irqN < EXT_IRQ_COUNT));
668 #else
669  EFM_ASSERT((irqN >= SVCall_IRQn) && (irqN < EXT_IRQ_COUNT));
670 #endif
671 
672  if (__get_PRIMASK() & 1) {
673  return true; // All IRQ's are disabled
674  }
675 
676  if (CORE_NvicIRQDisabled(irqN)) {
677  return true; // The IRQ in question is disabled
678  }
679 
680  irqPri = NVIC_GetPriority(irqN);
681 #if (__CORTEX_M >= 3)
682  basepri = __get_BASEPRI();
683  if ((basepri != 0)
684  && (irqPri >= (basepri >> (8 - __NVIC_PRIO_BITS)))) {
685  return true; // The IRQ in question has too low
686  } // priority vs. BASEPRI
687 #endif
688 
689  // Check if already in an interrupt handler, if so an interrupt with
690  // higher priority (lower priority value) can preempt.
691  activeIrq = (SCB->ICSR & SCB_ICSR_VECTACTIVE_Msk) >> SCB_ICSR_VECTACTIVE_Pos;
692  if ((activeIrq != 0)
693  && (irqPri >= NVIC_GetPriority((IRQn_Type)(activeIrq - 16)))) {
694  return true; // The IRQ in question has too low
695  } // priority vs. current active IRQ
696 
697  return false;
698 }

854 void CORE_InitNvicVectorTable(uint32_t *sourceTable,
855  uint32_t sourceSize,
856  uint32_t *targetTable,
857  uint32_t targetSize,
858  void *defaultHandler,
859  bool overwriteActive)
860 {
861  uint32_t i;
862 
863  // ASSERT on non SRAM based target table.
864  EFM_ASSERT(((uint32_t)targetTable >= RAM_MEM_BASE)
865  && ((uint32_t)targetTable < (RAM_MEM_BASE + RAM_MEM_SIZE)));
866 
867  // ASSERT if misaligned with respect to VTOR register implementation.
868 #if defined(SCB_VTOR_TBLBASE_Msk)
869  EFM_ASSERT(((uint32_t)targetTable & ~(SCB_VTOR_TBLOFF_Msk
870  | SCB_VTOR_TBLBASE_Msk)) == 0);
871 #else
872  EFM_ASSERT(((uint32_t)targetTable & ~SCB_VTOR_TBLOFF_Msk) == 0);
873 #endif
874 
875  // ASSERT if misaligned with respect to vector table size.
876  // Vector table address must be aligned at its size rounded up to nearest 2^n.
877  EFM_ASSERT(((uint32_t)targetTable
878  & ((1 << (32 - __CLZ((targetSize * 4) - 1))) - 1)) == 0);
879 
880  for (i=0; i<targetSize; i++) {
881  if (overwriteActive) { // Overwrite target entries ?
882  if (i<sourceSize) { // targetSize <= sourceSize
883  targetTable[i] = sourceTable[i];
884  } else { // targetSize > sourceSize
885  targetTable[i] = (uint32_t)defaultHandler;
886  }
887  } else { // Overwrite target entries which are 0
888  if (i<sourceSize) { // targetSize <= sourceSize
889  if (targetTable[i] == 0) {
890  targetTable[i] = sourceTable[i];
891  }
892  } else { // targetSize > sourceSize
893  if (targetTable[i] == 0) {
894  targetTable[i] = (uint32_t)defaultHandler;
895  }
896  }
897  }
898  }
899  SCB->VTOR = (uint32_t)targetTable;
900 }
