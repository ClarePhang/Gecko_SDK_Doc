109 void TEMPDRV_IRQHandler(void)
110 {
111  uint32_t flags = EMU_IntGetEnabled();
112  TEMPDRV_Callback_t activeCallback;
113  // High EMU value means a decreasing temperature
114  if (flags & EMU_IF_TEMPHIGH)
115  {
116  // High EMU interrupt = Low temp limit
117  if (lowCallback->callback != NULL)
118  {
119  activeCallback = lowCallback->callback;
120  memset(lowCallback, 0, sizeof(TEMPDRV_CallbackSet_t));
121  activeCallback(TEMPDRV_GetTemp(), TEMPDRV_LIMIT_LOW);
122  }
123  EMU_IntClear(EMU_IFC_TEMPHIGH);
124  }
125  else if (flags & EMU_IF_TEMPLOW)
126  {
127  // Low EMU interrupt = high temp limit

131  memset(highCallback, 0, sizeof(TEMPDRV_CallbackSet_t));

133  }
134  EMU_IntClear(EMU_IFC_TEMPLOW);

467 static void updateInterrupts(void)
468 {
469  // Find lowest temperature active high callback
470  uint8_t index;
471  for (index = 0; index < TEMPDRV_CALLBACK_DEPTH; index++)
472  {
473  // filter out only entries with valid callbacks
474  if (tempdrvHighCallbacks[index].callback!=NULL)
475  {
476  if (highCallback->callback == NULL)
477  {
478  highCallback = &tempdrvHighCallbacks[index];
479  }
480  else
481  {
482  if (tempdrvHighCallbacks[index].temp>highCallback->temp)
483  {
484  highCallback = &tempdrvHighCallbacks[index];
485  }
486  }
487  }
488  }
489  // If active callback, set and enable interrupt
490  if (highCallback->callback != NULL)
491  {
492  // EMU and TEMP are inversely proportional
493  EMU->TEMPLIMITS &= ~_EMU_TEMPLIMITS_TEMPLOW_MASK;
494  EMU->TEMPLIMITS |= highCallback->temp << _EMU_TEMPLIMITS_TEMPLOW_SHIFT;
495  EMU_IntEnable(EMU_IEN_TEMPLOW);
496  }
497  else
498  {
499  EMU_IntDisable(EMU_IEN_TEMPLOW);
500  }
501 
502  // Find highest temperature active low callback
503  for (index = 0; index < TEMPDRV_CALLBACK_DEPTH; index++)
504  {
505  // filter out only entries with valid callbacks
506  if (tempdrvLowCallbacks[index].callback!=NULL)
507  {
508  if (lowCallback->callback == NULL)
509  {
510  lowCallback = &tempdrvLowCallbacks[index];
511  }
512  else
513  {
514  if (tempdrvLowCallbacks[index].temp<lowCallback->temp)
515  {
516  lowCallback = &tempdrvLowCallbacks[index];
517  }
518  }
519  }
520  }
521  // If active callback, set and enable interrupt
522  if (lowCallback->callback!=NULL)
523  {
524  // EMU and TEMP are inversely proportional
525  EMU->TEMPLIMITS &= ~_EMU_TEMPLIMITS_TEMPHIGH_MASK;
526  EMU->TEMPLIMITS |= lowCallback->temp << _EMU_TEMPLIMITS_TEMPHIGH_SHIFT;
527  EMU_IntEnable(EMU_IEN_TEMPHIGH);
528  }
529  else
530  {
531  EMU_IntDisable(EMU_IEN_TEMPHIGH);
532  }
533 }

548 Ecode_t TEMPDRV_Init(void)
549 {
550  uint32_t DItemp, DIemu;
551 
552  // Flag up
553  TEMPDRV_InitState = true;
554 
555  // reset stack state by erasing callbacks
556  memset(tempdrvHighCallbacks, 0, sizeof (TEMPDRV_CallbackSet_t) * TEMPDRV_CALLBACK_DEPTH);
557  memset(tempdrvLowCallbacks, 0, sizeof (TEMPDRV_CallbackSet_t) * TEMPDRV_CALLBACK_DEPTH);
558  highCallback = &nullCallback;
559  lowCallback = &nullCallback;
560 
561  // Retrieve calibration data from DI page
562  DItemp = ((DEVINFO->CAL & _DEVINFO_CAL_TEMP_MASK) >> _DEVINFO_CAL_TEMP_SHIFT);
563  DIemu = ((DEVINFO->EMUTEMP & _DEVINFO_EMUTEMP_EMUTEMPROOM_MASK) >> _DEVINFO_EMUTEMP_EMUTEMPROOM_SHIFT);
564 
565  if ((DItemp == (_DEVINFO_CAL_TEMP_MASK >> _DEVINFO_CAL_TEMP_SHIFT))
566  || (DIemu == (_DEVINFO_EMUTEMP_EMUTEMPROOM_MASK >> _DEVINFO_EMUTEMP_EMUTEMPROOM_SHIFT)))
567  {
568  // Missing DI page calibration data, substitute fixed values
569  DItemp = fallbackTEMP;
570  DIemu = fallbackEMU;
571  }
572 
573  // calculate conversion offsets. Based on assumed slope of 5/8
574  calibrationEMU = (DIemu) + ((5*(DItemp))/8);
575  calibrationTEMP = (DItemp) + (8*(DIemu)/5);
576 
577  errataInit();
578 
579  disableInterrupts();
580  NVIC_ClearPendingIRQ(EMU_IRQn);
581  NVIC_EnableIRQ(EMU_IRQn);
582  updateInterrupts();
583 
584  return ECODE_EMDRV_TEMPDRV_OK;
585 }

751 Ecode_t TEMPDRV_RegisterCallback(int8_t temp,
752  TEMPDRV_LimitType_t limit,
753  TEMPDRV_Callback_t callback)
754 {
755  TEMPDRV_CallbackSet_t *set;
756  if (TEMPDRV_InitState == false)
757  {
758  return ECODE_EMDRV_TEMPDRV_NO_INIT;
759  }
760  // cannot register null callback
761  if (callback == NULL)
762  {
763  return ECODE_EMDRV_TEMPDRV_PARAM_ERROR;
764  }
765  if (limit == TEMPDRV_LIMIT_HIGH)
766  {
767  // current temperature is already higher than requested temperature
768  if (TEMPDRV_GetTemp() > temp)
769  {
770  return ECODE_EMDRV_TEMPDRV_BAD_LIMIT;
771  }
772  // Define callback set
773  set = tempdrvHighCallbacks;
774  }
775  else if (limit == TEMPDRV_LIMIT_LOW)
776  {
777  // current temperature is already lower than requested temperature
778  if (TEMPDRV_GetTemp() < temp)
779  {
780  return ECODE_EMDRV_TEMPDRV_BAD_LIMIT;
781  }
782  // Define callback set
783  set = tempdrvLowCallbacks;
784  }
785  else
786  {
787  // Invalid limit
788  return ECODE_EMDRV_TEMPDRV_PARAM_ERROR;
789  }
790 
791  // Cannot register duplicate temperature callback
792  if (checkForDuplicates(set, temp) == true)
793  {
794  return ECODE_EMDRV_TEMPDRV_DUP_TEMP;
795  }
796 
797  return addCallback(set, temp, callback);
798 }
